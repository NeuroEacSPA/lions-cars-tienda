    <script>
        const techSpec = document.getElementById('techSpec');
        const zoomToggle = document.getElementById('zoomToggle');
        const zoomLens = document.getElementById('zoomLens');
        const mainVehicle = document.getElementById('mainVehicle');
        const imageContainer = document.getElementById('imageContainer');
        const coordDisplay = document.getElementById('coordDisplay');
        
        let isZoomActive = false;

        function toggleTechSpec() {
            techSpec.classList.toggle('open');
        }

        zoomToggle.addEventListener('click', () => {
            isZoomActive = !isZoomActive;
            zoomToggle.innerHTML = isZoomActive ? 'ZOOM DE PRECISIÓN [ON]' : 'ZOOM DE PRECISIÓN [OFF]';
            zoomToggle.classList.toggle('border-red-500');
            zoomToggle.classList.toggle('text-red-500');
            zoomLens.style.display = isZoomActive ? 'block' : 'none';
        });

        imageContainer.addEventListener('mousemove', (e) => {
            const rect = mainVehicle.getBoundingClientRect();
            const x = e.clientX - rect.left;
            const y = e.clientY - rect.top;

            coordDisplay.innerText = `COORDS: X-${Math.round(x)} Y-${Math.round(y)}`;

            if (isZoomActive) {
                zoomLens.style.left = `${e.clientX - 75}px`;
                zoomLens.style.top = `${e.clientY - 75}px`;

                const ratio = 2.5;
                zoomLens.style.backgroundImage = `url('${mainVehicle.src}')`;
                zoomLens.style.backgroundSize = `${mainVehicle.width * ratio}px ${mainVehicle.height * ratio}px`;
                zoomLens.style.backgroundPosition = `-${x * ratio - 75}px -${y * ratio - 75}px`;
            }
        });

        // Simular carga de datos
        console.log("Iniciando Bio-Metric Interface...");
        setTimeout(() => {
            document.querySelector('.scanner-overlay').style.animation = "none";
        }, 3000);
    </script>
